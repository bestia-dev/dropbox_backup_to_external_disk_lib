27pub struct PathStructInSshFolder {
28    file_name: String,
29    tilde_file_path: CrossPathBuf,
30}
31
32impl PathStructInSshFolder {
33    /// Private key file path: CrossPathBuf have tilde. Later it transforms to full path.
34    pub fn new(file_name: String) -> Result<Self> {
35        let tilde_file_path = CrossPathBuf::new(&format!("~/.ssh/{file_name}"))?;
36        Ok(PathStructInSshFolder {
37            file_name,
38            tilde_file_path,
39        })
40    }
41
42    /// Get reference to file name.
43    pub fn get_file_name(&self) -> &str {
44        &self.file_name
45    }
46
47    /// Get reference to tilde file path CrossPathBuf.
48    pub fn get_cross_path(&self) -> &CrossPathBuf {
49        &self.tilde_file_path
50    }
51
52    /// Get full file path.
53    pub fn get_full_file_path(&self) -> std::path::PathBuf {
54        self.tilde_file_path.to_path_buf_current_os()
55    }
56
57    /// Return true if file exists on path.
58    pub fn exists(&self) -> bool {
59        self.tilde_file_path.exists()
60    }
61}
62
63impl std::fmt::Display for PathStructInSshFolder {
64    fn fmt(&self, f: &mut std::fmt::Formatter) -> std::fmt::Result {
65        write!(f, "{}", self.tilde_file_path)
66    }
67}

79/// Generate a random seed.
80///
81/// This seed will be signed with the private key and
82/// that will be the passcode for symmetric encryption
83/// We will need the bytes and the string representation
84pub(crate) fn random_seed_32bytes_and_string() -> Result<([u8; 32], String)> {
85    let mut seed_32bytes = [0_u8; 32];
86    use aes_gcm::aead::rand_core::RngCore;
87    aes_gcm::aead::OsRng.fill_bytes(&mut seed_32bytes);
88    let plain_seed_string = encode64_from_32bytes_to_string(seed_32bytes)?;
89    Ok((seed_32bytes, plain_seed_string))
90}

92/// Get the string from the file that is base64 encoded.
93///
94/// It is encoded just to obscure it a little.
95pub(crate) fn open_file_b64_get_string(plain_file_b64_path: &CrossPathBuf) -> Result<String> {
96    if !plain_file_b64_path.exists() {
97        return Err(Error::ErrorFromStr(
98            "{RED}Error: File {plain_file_b64_path} does not exist! {RESET}",
99        ));
100    }
101
102    let plain_file_text = plain_file_b64_path.read_to_string()?;
103    // it is encoded just to obscure it a little
104    let plain_file_text = decode64_from_string_to_string(&plain_file_text)?;
105
106    Ok(plain_file_text)
107}

109/// Shorten the `Vec<u8> to [u8;32]`.  
110pub(crate) fn shorten_vec_bytes_to_32bytes(vec_u8: Vec<u8>) -> Result<[u8; 32]> {
111    if vec_u8.len() < 32 {
112        return Err(Error::ErrorFromStr("The bytes must never be less then 32 bytes."));
113    }
114    let mut secret_passcode_32bytes = [0u8; 32];
115    secret_passcode_32bytes.copy_from_slice(&vec_u8[0..32]);
116
117    Ok(secret_passcode_32bytes)
118}

122/// Decode base64 from string to 32bytes.
123pub(crate) fn encode64_from_32bytes_to_string(bytes_32bytes: [u8; 32]) -> Result<String> {
124    Ok(<base64ct::Base64 as base64ct::Encoding>::encode_string(&bytes_32bytes))
125}

127/// Decode base64 from string to 32bytes.
128pub(crate) fn decode64_from_string_to_32bytes(plain_seed_string: &str) -> Result<[u8; 32]> {
129    let plain_seed_bytes = <base64ct::Base64 as base64ct::Encoding>::decode_vec(plain_seed_string)?;
130    let plain_seed_bytes_32bytes = shorten_vec_bytes_to_32bytes(plain_seed_bytes)?;
131    Ok(plain_seed_bytes_32bytes)
132}

134/// Encode base64 from bytes to string.
135pub(crate) fn encode64_from_bytes_to_string(plain_seed_bytes_32bytes: Vec<u8>) -> String {
136    <base64ct::Base64 as base64ct::Encoding>::encode_string(&plain_seed_bytes_32bytes)
137}

146/// Decode base64 from string to string.

158/// Returns the secret signed seed.
159///
160/// First it tries to use the ssh-agent.  
161/// Else it uses the private key and ask the user to input the passphrase.  
162/// If the passphrase is 'empty string' it will try ssh-agent one more time.
163/// The secret signed seed will be the actual password for symmetrical encryption.  
164/// Returns secret_password_bytes.  
165pub(crate) fn sign_seed_with_ssh_agent_or_private_key_file(
166    private_key_path_struct: &PathStructInSshFolder,
167    plain_seed_bytes_32bytes: [u8; 32],
168) -> Result<SecretBox<[u8; 32]>> {
169    let secret_passcode_32bytes_maybe = sign_seed_with_ssh_agent(plain_seed_bytes_32bytes, private_key_path_struct);
170    let secret_passcode_32bytes: SecretBox<[u8; 32]> = if secret_passcode_32bytes_maybe.is_ok() {
171        secret_passcode_32bytes_maybe?
172    } else {
173        // ask user to think about adding key into ssh-agent with ssh-add
174        println!("  {YELLOW}SSH key for encrypted secret_token is not found in the ssh-agent.{RESET}");
175        println!("  {YELLOW}Without ssh-agent, you will have to type the private key passphrase every time.{RESET}");
176        println!("  {YELLOW}This is more secure, but inconvenient.{RESET}");
177        println!("  {YELLOW}WARNING: using ssh-agent is less secure, because there is no need for user interaction.{RESET}");
178        println!("  {YELLOW}Knowing this, you can manually add the SSH private key to ssh-agent for 1 hour.{RESET}");
179        println!("  {YELLOW}You can simply open a new bash terminal and add the key there right now:{RESET}");
180        println!("{GREEN}ssh-add -t 1h {private_key_path_struct}{RESET}");
181        println!("  {YELLOW}Unlock the private key to decrypt the saved file.{RESET}");
182
183        match sign_seed_with_private_key_file(plain_seed_bytes_32bytes, &private_key_path_struct.tilde_file_path) {
184            Ok(secret_passcode_32bytes) => secret_passcode_32bytes,
185            Err(err) => {
186                if err.to_string() == "Passphrase empty" {
187                    // try with ssh-agent, because maybe the developer has ssh-add in another terminal right now
188                    return sign_seed_with_ssh_agent(plain_seed_bytes_32bytes, private_key_path_struct);
189                }
190                return Err(err);
191            }
192        }
193    };
194    Ok(secret_passcode_32bytes)
195}

197/// Sign seed with ssh-agent into 32 bytes secret.
198///
199/// This will be the true passcode for symmetrical encryption and decryption.  
200/// Returns secret_password_bytes.  
201fn sign_seed_with_ssh_agent(
202    plain_seed_bytes_32bytes: [u8; 32],
203    private_key_path_struct: &PathStructInSshFolder,
204) -> Result<SecretBox<[u8; 32]>> {
205    /// Internal function returns the public_key inside ssh-add
206    fn public_key_from_ssh_agent(
207        client: &mut ssh_agent_client_rs_git_bash::Client,
208        fingerprint_from_file: &str,
209    ) -> Result<ssh_key::PublicKey> {
210        let vec_identities = client
211            .list_all_identities()
212            .map_err(|err| Error::ErrorFromString(err.to_string()))?;
213        for identity in vec_identities.iter() {
214            if let ssh_agent_client_rs_git_bash::Identity::PublicKey(public_key) = identity {
215                let fingerprint_from_agent = public_key.key_data().fingerprint(Default::default()).to_string();
216
217                if fingerprint_from_agent == fingerprint_from_file {
218                    return Ok(public_key.clone().into_owned());
219                }
220            }
221        }
222        Err(Error::ErrorFromStr("This private key is not added to ssh-agent."))
223    }
224    let public_key_path_struct = PathStructInSshFolder::new(format!("{}.pub", private_key_path_struct.get_file_name()))?;
225    let public_key = ssh_key::PublicKey::read_openssh_file(&public_key_path_struct.get_full_file_path())?;
226    let fingerprint_from_file = public_key.fingerprint(Default::default()).to_string();
227
228    println!("  {YELLOW}Connect to ssh-agent on SSH_AUTH_SOCK{RESET}");
229    let var_ssh_auth_sock = std::env::var("SSH_AUTH_SOCK")?;
230    let path_ssh_auth_sock = CrossPathBuf::new(&var_ssh_auth_sock)?;
231    // import trait into scope
232    use ssh_agent_client_rs_git_bash::GitBash;
233    let mut ssh_agent_client =
234        ssh_agent_client_rs_git_bash::Client::connect_to_git_bash_or_linux(&path_ssh_auth_sock.to_path_buf_current_os())
235            .map_err(|err| Error::ErrorFromString(err.to_string()))?;
236
237    let public_key = public_key_from_ssh_agent(&mut ssh_agent_client, &fingerprint_from_file)?;
238
239    let mut secret_passcode_32bytes = SecretBox::new(Box::new([0u8; 32]));
240    // sign with public key from ssh-agent
241    // only the data part of the signature goes into as_bytes.
242    secret_passcode_32bytes.expose_secret_mut().copy_from_slice(
243        &ssh_agent_client
244            .sign(&public_key, &plain_seed_bytes_32bytes)
245            .map_err(|err| Error::ErrorFromString(err.to_string()))?
246            .as_bytes()[0..32],
247    );
248
249    Ok(secret_passcode_32bytes)
250}

252/// Sign the seed with the private key into 32 bytes secret.
253///
254/// User must input the passphrase to unlock the private key file.  
255/// This will be the true passcode for symmetrical encryption and decryption.  
256/// Returns secret_password_bytes.  
257fn sign_seed_with_private_key_file(
258    plain_seed_bytes_32bytes: [u8; 32],
259    private_key_file_path: &CrossPathBuf,
260) -> Result<SecretBox<[u8; 32]>> {
261    /// Internal function for user input passphrase
262    fn user_input_secret_passphrase() -> Result<SecretString> {
263        println!();
264        println!("{BLUE}Enter the passphrase for the SSH private key:{RESET}");
265
266        let secret_passphrase = SecretString::from(
267            inquire::Password::new("")
268                .without_confirmation()
269                .with_display_mode(inquire::PasswordDisplayMode::Masked)
270                .prompt()?,
271        );
272        if secret_passphrase.expose_secret().is_empty() {
273            return Err(Error::ErrorFromStr("Passphrase empty"));
274        }
275        Ok(secret_passphrase)
276    }
277    // the user is the only one that knows the passphrase to unlock the private key
278    let secret_user_passphrase: SecretString = user_input_secret_passphrase()?;
279
280    // sign_with_ssh_private_key_file
281    println!("  {YELLOW}Use ssh private key from file {RESET}");
282    let private_key = ssh_key::PrivateKey::read_openssh_file(&private_key_file_path.to_path_buf_current_os())?;
283    println!("  {YELLOW}Unlock the private key {RESET}");
284
285    // cannot use secrecy: PrivateKey does not have trait Zeroize
286    let mut secret_private_key = private_key.decrypt(secret_user_passphrase.expose_secret())?;
287
288    // FYI: this type of signature is compatible with ssh-agent because it does not involve namespace
289    println!("  {YELLOW}Sign the seed {RESET}");
290
291    let mut secret_passcode_32bytes = SecretBox::new(Box::new([0u8; 32]));
292    // only the data part of the signature goes into as_bytes.
293    // only the first 32 bytes
294    secret_passcode_32bytes
295        .expose_secret_mut()
296        .copy_from_slice(&rsa::signature::SignerMut::try_sign(&mut secret_private_key, &plain_seed_bytes_32bytes)?.as_bytes()[0..32]);
297
298    Ok(secret_passcode_32bytes)
299}

305/// Encrypts symmetrically secret_string_to_encrypt with secret_passcode_32bytes.
306///
307/// Consumes the secret_passcode_32bytes and secret_string_to_encrypt.  
308/// Returns the plain_encrypted_string, it is not a secret anymore.  
309pub(crate) fn encrypt_symmetric(secret_passcode_32bytes: SecretBox<[u8; 32]>, secret_string_to_encrypt: SecretString) -> Result<String> {
310    // nonce is salt
311    let nonce = <aes_gcm::Aes256Gcm as aes_gcm::AeadCore>::generate_nonce(&mut aes_gcm::aead::OsRng);
312    let Ok(cipher_text_encrypted) = aes_gcm::aead::Aead::encrypt(
313        // cipher_secret is the true passcode, here I don't know how to use secrecy, because the type has not the trait Zeroize
314        &<aes_gcm::Aes256Gcm as aes_gcm::KeyInit>::new(secret_passcode_32bytes.expose_secret().into()),
315        &nonce,
316        secret_string_to_encrypt.expose_secret().as_bytes(),
317    ) else {
318        panic!("{RED}Error: Encryption failed. {RESET}");
319    };
320
321    let mut encrypted_bytes = nonce.to_vec();
322    encrypted_bytes.extend_from_slice(&cipher_text_encrypted);
323    // plain encrypted string is not a secret anymore
324    let plain_encrypted_string = encode64_from_bytes_to_string(encrypted_bytes);
325
326    Ok(plain_encrypted_string)
327}

329/// Decrypts plain_encrypted_string with secret_passcode_32bytes.
330///
331/// Consumes secret_passcode_32bytes and encrypted_string.  
332/// Returns the secret_decrypted_string.  
333pub(crate) fn decrypt_symmetric(secret_passcode_32bytes: SecretBox<[u8; 32]>, plain_encrypted_string: String) -> Result<SecretString> {
334    let encrypted_bytes = <base64ct::Base64 as base64ct::Encoding>::decode_vec(&plain_encrypted_string)?;
335    // nonce is salt
336    let nonce = rsa::sha2::digest::generic_array::GenericArray::from_slice(&encrypted_bytes[..12]);
337    let cipher_text = &encrypted_bytes[12..];
338
339    let Ok(secret_decrypted_bytes) = aes_gcm::aead::Aead::decrypt(
340        // cipher_secret is the true passcode, here I don't know how to use secrecy, because the type has not the trait Zeroize
341        &<aes_gcm::Aes256Gcm as aes_gcm::KeyInit>::new(secret_passcode_32bytes.expose_secret().into()),
342        nonce,
343        cipher_text,
344    ) else {
345        panic!("{RED}Error: Decryption failed. {RESET}");
346    };
347    let secret_decrypted_string = SecretString::from(String::from_utf8(secret_decrypted_bytes)?);
348
349    Ok(secret_decrypted_string)
350}
