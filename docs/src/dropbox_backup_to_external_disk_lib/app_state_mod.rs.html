23/// This trait defines what functions must the bin project implement then the lib project can use them.  \
24///
25/// All IO must be defined inside the bin project: UI, env, file access.  \
26/// That way the same lib project can be used from different bin: CLI, TUI, GUI, env, file, network,...  \
27/// These methods will be available globally.  
28pub trait AppStateMethods: Sync + Send {
29    /// get encrypted authorization token from env var
30    fn load_keys_from_io(&self) -> Result<(String, String)>;
31    /// reference to app_config data
32    fn ref_app_config(&self) -> &AppConfig;
33    /// get locked Mutex
34    fn lock_proba(&self) -> std::sync::MutexGuard<'_, String>;
35}
36
37/// Global variable to store the Application state.  \
38///
39/// Global variables are so complicated in Rust.  \
40/// Read more: <https://www.sitepoint.com/rust-global-variables/>  \
41/// I will use Multi-threaded Global Variable with Runtime Initialization and Interior Mutability, the most complicated and usable one.  \
42/// All fields are private. Only the methods can be used globally.  \
43/// Example how to use it:  \
44///
45/// ```rust ignore
46/// fn global_app_state() -> &'static Box<dyn lib::AppStateMethods> {
47///     lib::APP_STATE.get().expect("Error OnceCell must not be empty.")
48/// }
49/// println!("{}", global_app_state().ref_app_config().path_list_for_create_folders);
50/// println!("{}", global_app_state().lock_proba());  
51/// ```
52pub static APP_STATE: once_cell::sync::OnceCell<Box<dyn AppStateMethods>> = once_cell::sync::OnceCell::new();
53
54pub fn global_config() -> &'static AppConfig {
55    global_app_state().ref_app_config()
56}

58/// Returns the global app state.
59// Clippy warning here, but I don't know how to change this.

61#[allow(clippy::borrowed_box)]

63    APP_STATE.get().expect("Bug: OnceCell must not be empty.")
