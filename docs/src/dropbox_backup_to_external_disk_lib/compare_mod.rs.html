28/// Compare list: the lists must be already sorted for this to work correctly.  
29fn compare_lists_internal(
30    ui_tx: std::sync::mpsc::Sender<String>,
31    base_path: &CrossPathBuf,
32    path_list_source_files: &CrossPathBuf,
33    path_list_destination_files: &CrossPathBuf,
34    path_list_for_download: &CrossPathBuf,
35    path_list_for_trash: &CrossPathBuf,
36    path_list_for_change_time_files: &CrossPathBuf,
37) -> Result<()> {
38    let file_list_source_files = FileTxt::open_for_read(path_list_source_files)?;
39    let string_list_source_files = file_list_source_files.read_to_string()?;
40    let vec_list_source_files: Vec<&str> = string_list_source_files.lines().collect();
41    println_to_ui_thread(
42        &ui_tx,
43        format!("{}: {}", file_list_source_files.file_name()?, vec_list_source_files.len()),
44    );
45
46    let file_list_destination_files = FileTxt::open_for_read(path_list_destination_files)?;
47    let string_list_destination_files = file_list_destination_files.read_to_string()?;
48    let vec_list_destination_files: Vec<&str> = string_list_destination_files.lines().collect();
49    println_to_ui_thread(
50        &ui_tx,
51        format!("{}: {}", file_list_destination_files.file_name()?, vec_list_destination_files.len()),
52    );
53
54    let mut vec_for_download: Vec<String> = vec![];
55    let mut vec_for_trash: Vec<String> = vec![];
56    let mut vec_for_change_time_files: Vec<String> = vec![];
57    let mut cursor_source = 0;
58    let mut cursor_destination = 0;
59    //avoid making new allocations or shadowing inside a loop
60    let mut vec_line_destination: Vec<&str> = vec![];
61    let mut vec_line_source: Vec<&str> = vec![];
62    let mut i = 0;
63    loop {
64        vec_line_destination.truncate(3);
65        vec_line_source.truncate(3);
66
67        if cursor_source >= vec_list_source_files.len() && cursor_destination >= vec_list_destination_files.len() {
68            break;
69        } else if cursor_source >= vec_list_source_files.len() {
70            // final lines
71            vec_for_trash.push(vec_list_destination_files[cursor_destination].to_string());
72            cursor_destination += 1;
73        } else if cursor_destination >= vec_list_destination_files.len() {
74            // final lines
75            vec_for_download.push(vec_list_source_files[cursor_source].to_string());
76            cursor_source += 1;
77        } else {
78            //compare the 2 lines
79            // /Video_Backup/DVDs/BikeManual/om/FOXHelp/jap/float_x.htm	2007-01-08T19:31:44Z	45889   content_hash
80            vec_line_source = vec_list_source_files[cursor_source].split("\t").collect();
81            vec_line_destination = vec_list_destination_files[cursor_destination].split("\t").collect();
82            // UncasedStr preserves the case in the string, but comparison is done case insensitive
83            let path_source: &UncasedStr = vec_line_source[0].into();
84            let path_destination: &UncasedStr = vec_line_destination[0].into();
85
86            if path_source.lt(path_destination) {
87                vec_for_download.push(vec_list_source_files[cursor_source].to_string());
88                cursor_source += 1;
89            } else if path_source.gt(path_destination) {
90                vec_for_trash.push(vec_list_destination_files[cursor_destination].to_string());
91                cursor_destination += 1;
92            } else if vec_line_source[2] != vec_line_destination[2] {
93                // equal names, different size
94                vec_for_download.push(vec_list_source_files[cursor_source].to_string());
95                cursor_destination += 1;
96                cursor_source += 1;
97            } else {
98                // equal names, equal size, check date and later check content_hash
99                let source_modified_dt_utc: DateTime<Utc> = DateTime::parse_from_rfc3339(vec_line_source[1])?.into();
100                let destination_modified_dt_utc: DateTime<Utc> = DateTime::parse_from_rfc3339(vec_line_destination[1])?.into();
101                // if date is more than 2 seconds different
102                // incredible, incredible, incredible. exFAT is a Microsoft disk format for external disks. It allows for 10ms resolution for LastWrite/modified datetime.
103                // But Microsoft in Win10 driver for exFAT uses only 2seconds resolution. Crazy! After 20 years of existence.
104                // this means that if the time difference is less then 2 seconds, they are probably the same file
105                if chrono::Duration::from(source_modified_dt_utc - destination_modified_dt_utc).abs() > chrono::Duration::seconds(2) {
106                    // 2025-09-21 another strange behavior: for some git object files the modified date is different on my local disk and on Dropbox
107                    // I don't know why is that, but I have 5000 of these files small and large. I suppose the content is equal therefor Dropbox does not sync them.
108                    // /BestiaDev/github_backup_active/github_backup_private/obsidian_bestia_dev/.git/objects/1f/55b1a1662d4c06e1909f73877513bf38cc390e
109                    // I can recognize them id the path contains '/.git/'
110                    // I will use content_hash to be sure that these files are equal.
111                    let path_global_to_destination_file = base_path.join_relative(vec_line_destination[0])?;
112                    // get content_hash from destination file
113                    let local_content_hash = format!(
114                        "{:x}",
115                        DropboxContentHasher::hash_file(path_global_to_destination_file.to_path_buf_current_os())?
116                    );
117
118                    // a simple rudimentary progress bar
119                    if i > 40 {
120                        i = 0;
121                        print!(".");
122                        // flush
123                        use std::io::Write;
124                        std::io::stdout().flush()?;
125                    }
126                    i += 1;
127
128                    if local_content_hash != vec_line_source[3] {
129                        // println!("{} {}", local_content_hash, vec_line_source[3]);
130                        vec_for_download.push(vec_list_source_files[cursor_source].to_string());
131                    } else {
132                        // these files need change datetime
133                        vec_for_change_time_files.push(vec_list_source_files[cursor_source].to_string());
134                    }
135                }
136                // else the metadata is the same, no action
137                cursor_destination += 1;
138                cursor_source += 1;
139            }
140        }
141    }
142    println!();
143
144    let mut file_list_for_change_time_files = FileTxt::open_for_read_and_write(path_list_for_change_time_files)?;
145    println_to_ui_thread(
146        &ui_tx,
147        format!(
148            "{}: {}",
149            file_list_for_change_time_files.file_name()?,
150            vec_for_change_time_files.len()
151        ),
152    );
153    let string_for_change_time_files = vec_for_change_time_files.join("\n");
154    file_list_for_change_time_files.write_append_str(&string_for_change_time_files)?;
155
156    let mut file_list_for_trash_files = FileTxt::open_for_read_and_write(path_list_for_trash)?;
157    println_to_ui_thread(
158        &ui_tx,
159        format!("{}: {}", file_list_for_trash_files.file_name()?, vec_for_trash.len()),
160    );
161    let string_for_trash_files = vec_for_trash.join("\n");
162    file_list_for_trash_files.write_append_str(&string_for_trash_files)?;
163
164    let mut file_list_for_downloads = FileTxt::open_for_read_and_write(path_list_for_download)?;
165    println_to_ui_thread(
166        &ui_tx,
167        format!("{}: {}", file_list_for_downloads.file_name()?, vec_for_download.len()),
168    );
169    let string_for_download = vec_for_download.join("\n");
170    file_list_for_downloads.write_append_str(&string_for_download)?;
171
172    Ok(())
173}

175/// Compare folders and write folders to trash into path_list_for_trash_folders.  \
176///
177/// The list is already sorted.  
178pub fn compare_folders(
179    ui_tx: std::sync::mpsc::Sender<String>,
180    string_list_source_folders: &str,
181    string_list_destination_folders: &str,
182    file_list_for_trash_folders: &mut FileTxt,
183    file_list_for_create_folders: &mut FileTxt,
184) -> Result<()> {
185    let vec_list_source_folders: Vec<&str> = string_list_source_folders.lines().collect();
186    let vec_list_destination_folders: Vec<&str> = string_list_destination_folders.lines().collect();
187
188    let mut vec_for_trash: Vec<String> = vec![];
189    file_list_for_trash_folders.empty()?;
190    let mut vec_for_create: Vec<String> = vec![];
191    file_list_for_create_folders.empty()?;
192    let mut cursor_source = 0;
193    let mut cursor_destination = 0;
194
195    loop {
196        if cursor_source >= vec_list_source_folders.len() && cursor_destination >= vec_list_destination_folders.len() {
197            // all lines are processed
198            break;
199        } else if cursor_destination >= vec_list_destination_folders.len() {
200            // final lines
201            vec_for_create.push(vec_list_source_folders[cursor_source].to_string());
202            cursor_source += 1;
203        } else if cursor_source >= vec_list_source_folders.len() {
204            // final lines
205            vec_for_trash.push(vec_list_destination_folders[cursor_destination].to_string());
206            cursor_destination += 1;
207        } else {
208            // compare the 2 lines
209            // UncasedStr preserves the case in the string, but comparison is done case insensitive
210            let path_source: &UncasedStr = vec_list_source_folders[cursor_source].into();
211            let path_destination: &UncasedStr = vec_list_destination_folders[cursor_destination].into();
212            if path_source.lt(path_destination) {
213                vec_for_create.push(vec_list_source_folders[cursor_source].to_string());
214                cursor_source += 1;
215            } else if path_source.gt(path_destination) {
216                vec_for_trash.push(vec_list_destination_folders[cursor_destination].to_string());
217                cursor_destination += 1;
218            } else {
219                // else no action, just increment cursors
220                cursor_destination += 1;
221                cursor_source += 1;
222            }
223        }
224    }
225    println_to_ui_thread(
226        &ui_tx,
227        format!("{}: {}", file_list_for_trash_folders.file_name()?, vec_for_trash.len()),
228    );
229    let string_for_trash_files = vec_for_trash.join("\n");
230    file_list_for_trash_folders.write_append_str(&string_for_trash_files)?;
231    println_to_ui_thread(
232        &ui_tx,
233        format!("{}: {}", file_list_for_create_folders.file_name()?, vec_for_create.len()),
234    );
235    let string_for_create = vec_for_create.join("\n");
236    file_list_for_create_folders.write_append_str(&string_for_create)?;
237    Ok(())
238}
