143/// shorten path for screen to avoid word-wrap
144pub fn shorten_string(text: &str, x_max_char: u16) -> String {
145    if text.chars().count() > x_max_char as usize {
146        let x_half_in_char = (x_max_char / 2 - 2) as usize;
147        let pos1_in_bytes = byte_pos_from_chars(text, x_half_in_char);
148        let pos2_in_bytes = byte_pos_from_chars(text, text.chars().count() - x_half_in_char);
149        format!("{}...{}", &text[..pos1_in_bytes], &text[pos2_in_bytes..])
150    } else {
151        text.to_string()
152    }
153}

155/// it is used for substring, because string slice are counted in bytes and not chars.
156/// if we have multi-byte unicode characters we can get an error if the boundary is not on char boundary.
157pub fn byte_pos_from_chars(text: &str, char_pos: usize) -> usize {
158    text.char_indices().nth(char_pos).unwrap().0
159}

161/// sort string lines case insensitive
162pub fn sort_string_lines(output_string: &str) -> String {
163    let mut sorted_local: Vec<&str> = output_string.lines().collect();
164
165    use rayon::prelude::*;
166    sorted_local.par_sort_unstable_by(|a, b| {
167        let aa: &UncasedStr = (*a).into();
168        let bb: &UncasedStr = (*b).into();
169        aa.cmp(bb)
170    });
171    // return
172    sorted_local.join("\n")
173}

175/// sort list case insensitive
176pub fn sort_list(mut list: Vec<String>) -> String {
177    use rayon::prelude::*;
178    list.par_sort_unstable_by(|a, b| {
179        let aa: &uncased::UncasedStr = a.as_str().into();
180        let bb: &uncased::UncasedStr = b.as_str().into();
181        aa.cmp(bb)
182    });
183    // join to string and write to file
184    list.join("\n")
185}
