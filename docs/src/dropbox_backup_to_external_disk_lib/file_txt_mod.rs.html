3use crossplatform_path::CrossPathBuf;
4
5/// my object to work with text files
6pub struct FileTxt {
7    file_path: CrossPathBuf,
8    file_txt: std::fs::File,
9}
10
11impl FileTxt {
12    /// if file not exist, returns error
13    pub fn open_for_read(path: &CrossPathBuf) -> std::io::Result<Self> {
14        let file = std::fs::File::options().read(true).open(path.to_path_buf_current_os())?;
15        Ok(FileTxt {
16            file_txt: file,
17            file_path: path.to_owned(),
18        })
19    }
20
21    /// if file not exist, it creates it
22    pub fn open_for_read_and_write(path: &CrossPathBuf) -> std::io::Result<Self> {
23        if !path.exists() {
24            std::fs::File::create(path.to_path_buf_current_os()).unwrap();
25        }
26        let file = std::fs::File::options()
27            .read(true)
28            .write(true)
29            .open(path.to_path_buf_current_os())?;
30
31        Ok(FileTxt {
32            file_txt: file,
33            file_path: path.to_owned(),
34        })
35    }
36
37    // returns file path
38    pub fn file_path(&self) -> &CrossPathBuf {
39        &self.file_path
40    }
41
42    // returns file name, just the last path fragment
43    pub fn file_name(&self) -> String {
44        match self.file_path.as_str().split("/").collect::<Vec<&str>>().last() {
45            Some(t) => t.to_string(),
46            None => self.file_path.as_str().to_string(),
47        }
48    }
49
50    /// This method is similar to fs::read_to_string, but instead of a path it expects a File parameter
51    /// So is possible to open a File in the bin part of the project and then pass it to the lib part of project.
52    /// All input and output should be in the bin part of project and not in the lib.
53    pub fn read_to_string(&self) -> Result<String, crossplatform_path::LibraryError> {
54        self.file_path.read_to_string()
55    }
56
57    /// write str to file (append)
58    pub fn write_append_str(&mut self, str: &str) -> std::io::Result<()> {
59        std::io::Write::write_all(&mut self.file_txt, str.as_bytes())?;
60        Ok(())
61    }
62
63    /// empty the file
64    pub fn empty(&mut self) -> std::io::Result<()> {
65        self.file_txt.set_len(0).unwrap();
66        Ok(())
67    }
68}
