61/// Get remote list in parallel.  \
62///
63/// First get the first level of folders and then request in parallel sub-folders recursively.  
64pub fn list_remote(
65    ui_tx: std::sync::mpsc::Sender<(String, ThreadName)>,
66    mut file_list_source_files: FileTxt,
67    mut file_list_source_folders: FileTxt,
68) -> Result<(), DropboxBackupToExternalDiskError> {
69    let list_remote_start = std::time::Instant::now();
70    // empty the files. I want all or nothing result here if the process is terminated prematurely.
71    file_list_source_files.empty()?;
72    file_list_source_folders.empty()?;
73
74    let token = get_authorization_token()?;
75    let client = dropbox_sdk::default_client::UserAuthDefaultClient::new(token.clone());
76    let client_ref = &client;
77    // walkdir non-recursive for the first level of folders
78    let (folder_list_root, file_list_root) = list_remote_folder(&client, "/", 0, false, ui_tx.clone())?;
79
80    // threadpool with 8 threads
81    let pool = rayon::ThreadPoolBuilder::new()
82        .num_threads(8)
83        .build()
84        .expect("Bug: rayon ThreadPoolBuilder");
85    pool.scope({
86        // Prepare variables to be moved/captured to the closure. All is isolated in a block scope.
87        let mut folder_list_all = vec![];
88        let mut file_list_all = file_list_root;
89        // channel for inter-thread communication to send folder/files lists
90        let (list_tx, list_rx) = std::sync::mpsc::channel();
91        // only the closure is actually spawned, because it is the return value of the block
92        move |scoped| {
93            // these folders will request walkdir recursive in parallel
94            // loop in a new thread, so the send msg will come immediately
95            for folder_path in folder_list_root.iter() {
96                // I store paths without leading slash, but it needs it for the API call.
97                let folder_path = format!("/{folder_path}");
98                // execute in a separate threads, or waits for a free thread from the pool
99                // scoped.spawn closure cannot return a Result<>, but it can send it as inter-thread message
100                scoped.spawn({
101                    // Prepare variables to be moved/captured to the closure. All is isolated in a block scope.
102                    // We are in a loop here. It means for every round we need new clones to move/capture/consume.
103                    let folder_path = folder_path.clone();
104                    let ui_tx_move_to_closure = ui_tx.clone();
105                    let ui_tx_move_to_closure_2 = ui_tx.clone();
106                    let list_tx_move_to_closure = list_tx.clone();
107                    // only the closure is actually spawned, because it is the return value of the block
108                    move |_| {
109                        let thread_num = rayon::current_thread_index().expect("Bug: rayon current_thread_index") as ThreadNum;
110                        // catch propagated errors and communicate errors to user or developer
111                        // spawned closure cannot propagate error with ?
112                        match list_remote_folder(client_ref, &folder_path, thread_num, true, ui_tx_move_to_closure) {
113                            Ok(folder_list_and_file_list) => {
114                                list_tx_move_to_closure.send(folder_list_and_file_list).expect("Bug: mpsc send")
115                            }
116                            Err(err) => println_to_ui_thread_with_thread_name(
117                                &ui_tx_move_to_closure_2,
118                                format!("Error in thread {err}"),
119                                &format!("R{thread_num}"),
120                            ),
121                        }
122                    }
123                });
124            }
125
126            // the receiver reads all msgs from the queue
127            drop(list_tx);
128            let mut all_folder_count = 0;
129            let mut all_file_count = 0;
130            for (folder_list, file_list) in &list_rx {
131                all_folder_count += folder_list.len();
132                all_file_count += file_list.len();
133                folder_list_all.extend_from_slice(&folder_list);
134                file_list_all.extend_from_slice(&file_list);
135            }
136
137            println_to_ui_thread_with_thread_name(&ui_tx, format!("Remote folder count: {all_folder_count}"), "R");
138            let string_folder_list = crate::utils_mod::sort_list(folder_list_all);
139            file_list_source_folders
140                .write_append_str(&string_folder_list)
141                .expect("Bug: file_list_source_folders must be writable");
142
143            println_to_ui_thread_with_thread_name(&ui_tx, format!("Remote file count: {all_file_count}"), "R");
144            let string_file_list = crate::utils_mod::sort_list(file_list_all);
145            file_list_source_files
146                .write_append_str(&string_file_list)
147                .expect("Bug: file_list_source_files must be writable");
148
149            println_to_ui_thread_with_thread_name(
150                &ui_tx,
151                format!("Remote duration in seconds: {}", list_remote_start.elapsed().as_secs()),
152                "R",
153            );
154        }
155    });
156
157    Ok(())
158}

160/// List remote folder.  
161pub fn list_remote_folder(
162    client: &dropbox_sdk::default_client::UserAuthDefaultClient,
163    path: &str,
164    thread_num: ThreadNum,
165    recursive: bool,
166    ui_tx: std::sync::mpsc::Sender<(String, ThreadName)>,
167) -> Result<FolderListAndFileList, DropboxBackupToExternalDiskError> {
168    let mut folder_list: FolderList = vec![];
169    let mut file_list: FileList = vec![];
170    let mut last_send_ms = std::time::Instant::now();
171
172    match dropbox_list_folder(client, path, recursive) {
173        Ok(Ok(iterator)) => {
174            for entry_result in iterator {
175                match entry_result {
176                    Ok(Ok(dropbox_sdk::files::Metadata::Folder(entry))) => {
177                        // path_display is not 100% case accurate. Dropbox is case-insensitive and preserves the casing only for the metadata_name, not path.
178                        let folder_path = entry.path_display.unwrap_or(entry.name);
179                        let folder_path = folder_path.trim_start_matches("/");
180                        // writing to screen is slow, I will not write every folder/file, but will wait for 100ms
181                        if last_send_ms.elapsed().as_millis() >= 100 {
182                            println_to_ui_thread_with_thread_name(
183                                &ui_tx,
184                                format!("Folder: {}", crate::shorten_string(folder_path, 80)),
185                                &format!("R{thread_num}"),
186                            );
187                            last_send_ms = std::time::Instant::now();
188                        }
189                        folder_list.push(folder_path.to_string());
190                    }
191                    Ok(Ok(dropbox_sdk::files::Metadata::File(entry))) => {
192                        // write csv tab delimited
193                        // avoid strange files *com.dropbox.attrs
194                        // path_display is not 100% case accurate. Dropbox is case-insensitive and preserves the casing only for the metadata_name, not path.
195                        let file_path = entry.path_display.unwrap_or(entry.name);
196                        let file_path = file_path.trim_start_matches("/");
197                        if !file_path.ends_with("com.dropbox.attrs") {
198                            // writing to screen is slow, I will not write every folder/file, but will wait for 100ms
199                            if last_send_ms.elapsed().as_millis() >= 100 {
200                                println_to_ui_thread_with_thread_name(
201                                    &ui_tx,
202                                    format!("File: {}", crate::shorten_string(file_path, 80)),
203                                    &format!("R{thread_num}"),
204                                );
205                                last_send_ms = std::time::Instant::now();
206                            }
207                            file_list.push(format!(
208                                "{}\t{}\t{}\t{}",
209                                file_path,
210                                entry.client_modified,
211                                entry.size,
212                                entry.content_hash.unwrap()
213                            ));
214                        }
215                    }
216                    Ok(Ok(dropbox_sdk::files::Metadata::Deleted(_entry))) => {
217                        return Err(DropboxBackupToExternalDiskError::ErrorFromString(format!(
218                            "R{thread_num} Error unexpected deleted entry"
219                        )));
220                    }
221                    Ok(Err(e)) => {
222                        return Err(DropboxBackupToExternalDiskError::ErrorFromString(format!(
223                            "R{thread_num} Error from files/list_folder_continue: {e}"
224                        )));
225                    }
226                    Err(e) => {
227                        return Err(DropboxBackupToExternalDiskError::ErrorFromString(format!(
228                            "R{thread_num} Error API request: {e}"
229                        )));
230                    }
231                }
232            }
233            // return FolderListAndFileList
234            Ok((folder_list, file_list))
235        }
236        Ok(Err(e)) => Err(DropboxBackupToExternalDiskError::ErrorFromString(format!(
237            "R{thread_num} Error from files/list_folder: {e}"
238        ))),
239        Err(e) => Err(DropboxBackupToExternalDiskError::ErrorFromString(format!(
240            "R{thread_num} Error API request: {e}"
241        ))),
242    }
243}

245/// Dropbox function to list folders.  
246fn dropbox_list_folder<'a>(
247    client: &'a dropbox_sdk::default_client::UserAuthDefaultClient,
248    path: &str,
249    recursive: bool,
250) -> dropbox_sdk::Result<Result<DirectoryIterator<'a>, dropbox_sdk::files::ListFolderError>> {
251    // validate input parameters
252    // assert! macro will panic if false.
253    // Is is not used for user input validation but to express an assumption (for code readers) and to catch coding bugs (in the caller).
254    // You express which invariant should always be true and therefore there should be no failure. If a failure happens it is a coding bug in the caller.
255    // An invariant is any "logical rule that must be obeyed" (assumption) that can be communicated to a human, but not to your compiler.
256    assert!(path.starts_with('/'), "Error path needs to be absolute (start with a '/')");
257
258    let path = if path == "/" {
259        // Root folder should be requested as empty string
260        String::new()
261    } else {
262        path.to_owned()
263    };
264
265    match dropbox_sdk::files::list_folder(client, &dropbox_sdk::files::ListFolderArg::new(path).with_recursive(recursive)) {
266        Ok(Ok(result)) => {
267            let cursor = if result.has_more { Some(result.cursor) } else { None };
268
269            Ok(Ok(DirectoryIterator {
270                client,
271                cursor,
272                buffer: result.entries.into(),
273            }))
274        }
275        Ok(Err(e)) => Ok(Err(e)),
276        Err(e) => Err(e),
277    }
278}

280/// Iterator for Directory on remote Dropbox storage.  
281struct DirectoryIterator<'a> {
282    client: &'a dropbox_sdk::default_client::UserAuthDefaultClient,
283    buffer: std::collections::VecDeque<dropbox_sdk::files::Metadata>,
284    cursor: Option<String>,
285}
286
287impl<'a> Iterator for DirectoryIterator<'a> {
288    type Item = dropbox_sdk::Result<Result<dropbox_sdk::files::Metadata, dropbox_sdk::files::ListFolderContinueError>>;
289
290    fn next(&mut self) -> Option<Self::Item> {
291        if let Some(entry) = self.buffer.pop_front() {
292            Some(Ok(Ok(entry)))
293        } else if let Some(cursor) = self.cursor.take() {
294            match dropbox_sdk::files::list_folder_continue(self.client, &dropbox_sdk::files::ListFolderContinueArg::new(cursor)) {
295                Ok(Ok(result)) => {
296                    self.buffer.extend(result.entries);
297                    if result.has_more {
298                        self.cursor = Some(result.cursor);
299                    }
300                    self.buffer.pop_front().map(|entry| Ok(Ok(entry)))
301                }
302                Ok(Err(e)) => Some(Ok(Err(e))),
303                Err(e) => Some(Err(e)),
304            }
305        } else {
306            None
307        }
308    }
309}

327/// Download files from list.  \
328///
329/// It removes just_downloaded from list_for_download, so this function can be stopped and then called again.  
330pub fn download_from_list(
331    ui_tx: std::sync::mpsc::Sender<(String, ThreadName)>,
332    ext_disk_base_path: &CrossPathBuf,
333    file_list_for_download: &mut FileTxt,
334    file_list_just_downloaded: &mut FileTxt,
335) -> Result<(), DropboxBackupToExternalDiskError> {
336    let list_for_download = file_list_for_download.read_to_string()?;
337    let mut vec_list_for_download: Vec<&str> = list_for_download.lines().collect();
338
339    //remove list_just_downloaded from list_for_download
340    let list_just_downloaded = file_list_just_downloaded.read_to_string()?;
341    if !list_just_downloaded.is_empty() {
342        let vec_list_just_downloaded: Vec<&str> = list_just_downloaded.lines().collect();
343        for just_downloaded in vec_list_just_downloaded.iter() {
344            vec_list_for_download.retain(|line| !line.starts_with(just_downloaded))
345        }
346        let string_for_download = vec_list_for_download.join("\n");
347        file_list_for_download.empty()?;
348        file_list_for_download.write_append_str(&string_for_download)?;
349        file_list_just_downloaded.empty()?;
350    }
351
352    download_from_vec(ui_tx, ext_disk_base_path, &mut vec_list_for_download, file_list_just_downloaded)?;
353
354    Ok(())
355}

357fn download_from_vec(
358    ui_tx: std::sync::mpsc::Sender<(String, ThreadName)>,
359    ext_disk_base_path: &CrossPathBuf,
360    vec_list_for_download: &mut Vec<&str>,
361    file_list_just_downloaded: &mut FileTxt,
362) -> Result<(), DropboxBackupToExternalDiskError> {
363    let token = get_authorization_token()?;
364    let client = dropbox_sdk::default_client::UserAuthDefaultClient::new(token);
365    // I have to create a reference before the move-closure. So the reference is moved to the closure and not the object.
366    let client_ref = &client;
367    // channel for inter-thread communication to send messages that will be appended to files
368    let (files_append_tx, files_append_rx) = std::sync::mpsc::channel();
369    //8 threads to download in parallel
370    let pool = rayon::ThreadPoolBuilder::new().num_threads(8).build().unwrap();
371    pool.scope(move |scoped| {
372        for line_path_to_download in vec_list_for_download.iter() {
373            // execute in 4 separate threads, or waits for a free thread from the pool
374            scoped.spawn({
375                // Prepare variables to be moved/captured to the closure. All is isolated in a block scope.
376                let line: Vec<&str> = line_path_to_download.split("\t").collect();
377                let path_to_download = CrossPathBuf::new(line[0]).expect("Error handling inside closures is not good.");
378                let ui_tx_clone = ui_tx.clone();
379                let ui_tx_move_to_closure_2 = ui_tx.clone();
380                let files_append_tx_move_to_closure = files_append_tx.clone();
381                // only the closure is actually spawned, because it is the return value of the block
382                move |_| {
383                    let thread_num = rayon::current_thread_index().expect("Bug: thread num must exist.");
384                    // catch propagated errors and communicate errors to user or developer
385                    // spawned closure cannot propagate error with ?
386                    match download_internal(
387                        ui_tx_clone,
388                        ext_disk_base_path,
389                        client_ref,
390                        thread_num as i32,
391                        &path_to_download,
392                        files_append_tx_move_to_closure,
393                    ) {
394                        Ok(()) => {}
395                        Err(err) => println_to_ui_thread_with_thread_name(
396                            &ui_tx_move_to_closure_2,
397                            format!("Error in thread {err}"),
398                            &format!("R{thread_num}"),
399                        ),
400                    }
401                }
402            });
403        }
404
405        // region: Receiver reads all msgs from the queue
406        // and them appends it neatly in files. Because only this thread writes to files there cannot be data race condition.
407        drop(files_append_tx);
408        for just_downloaded in &files_append_rx {
409            if !just_downloaded.is_empty() {
410                file_list_just_downloaded
411                    .write_append_str(&format!("{just_downloaded}\n"))
412                    .expect("Bug: file_list_just_downloaded must be writable.");
413            }
414        }
415        // endregion: Receiver reads all msgs from the queue
416    });
417
418    Ok(())
419}

421/// Download one file with client object dropbox_sdk::default_client::UserAuthDefaultClient.
422fn download_internal(
423    ui_tx: std::sync::mpsc::Sender<(String, ThreadName)>,
424    ext_disk_base_path: &CrossPathBuf,
425    client: &dropbox_sdk::default_client::UserAuthDefaultClient,
426    thread_num: i32,
427    path_to_download: &CrossPathBuf,
428    files_append_tx: std::sync::mpsc::Sender<String>,
429) -> Result<(), DropboxBackupToExternalDiskError> {
430    let thread_name = format!("R{thread_num}");
431    // add a leading slash to path_to_download because the dropbox_sdk needs it
432    let path_to_download = path_to_download.add_start_slash()?;
433    let local_path = ext_disk_base_path.join_relative(path_to_download.as_str())?;
434    local_path.create_dir_all_for_file()?;
435    let modified_str;
436    let metadata_size;
437    let mut just_downloaded = String::new();
438
439    // get datetime from remote
440
441    let get_metadata_arg = dropbox_sdk::files::GetMetadataArg::new(path_to_download.to_string());
442    let metadata = (dropbox_sdk::files::get_metadata(client, &get_metadata_arg)?)?;
443    match metadata {
444        dropbox_sdk::files::Metadata::File(metadata) => {
445            modified_str = metadata.client_modified;
446            metadata_size = metadata.size;
447        }
448        _ => {
449            return Err(DropboxBackupToExternalDiskError::ErrorFromStr("This is not a file on Dropbox"));
450        }
451    }
452
453    let system_time_modified = humantime::parse_rfc3339(&modified_str).expect("Bug: parse_rfc3339 must succeed");
454    // let modified = filetime::FileTime::from_system_time(system_time_modified);
455
456    // files of size 0 cannot be downloaded. I will just create them empty, because download empty file causes error 416
457    if metadata_size == 0 {
458        if local_path.exists() {
459            std::fs::remove_file(local_path.to_path_buf_current_os()).expect("Bug: remove file must succeed");
460        }
461        let _file = FileTxt::open_for_read_and_write(&local_path).expect("Bug: open_for_read_and_write must succeed.");
462        println_to_ui_thread_with_thread_name(&ui_tx, local_path.to_string(), &thread_name);
463    } else {
464        let mut bytes_out = 0u64;
465        let download_arg = dropbox_sdk::files::DownloadArg::new(path_to_download.to_string());
466        let base_temp_path_to_download = ext_disk_base_path.join_relative("0_backup_temp")?.join_relative("download_temp")?;
467        base_temp_path_to_download.create_dir_all()?;
468        let unique_name = local_path.as_str().replace("/", "_");
469        let temp_local_path = base_temp_path_to_download.join_relative(&unique_name)?;
470        let mut file = std::fs::OpenOptions::new()
471            .create(true)
472            .truncate(true)
473            .write(true)
474            .open(temp_local_path.to_path_buf_current_os())?;
475        file.set_modified(system_time_modified).unwrap();
476
477        // I will download to a temp folder and then move the file to the right folder only when the download is complete.
478        'download: loop {
479            // TODO: I want to press a key to stop the downloading gracefully
480            // but this thread is NOT the ui thread
481            let result = dropbox_sdk::files::download(client, &download_arg, Some(bytes_out), None);
482            match result {
483                Ok(Ok(download_result)) => {
484                    let mut body = download_result.body.expect("Bug: body must exist");
485                    loop {
486                        // limit read to 1 MiB per loop iteration so we can output progress
487                        // let mut input_chunk = (&mut body).take(1_048_576);
488                        use std::io::Read; //import trait
489                        let mut input_chunk = (&mut body).take(1_048_576);
490                        match std::io::copy(&mut input_chunk, &mut file) {
491                            Ok(0) => {
492                                break 'download;
493                            }
494                            Ok(len) => {
495                                bytes_out += len;
496                                if let Some(total) = download_result.content_length {
497                                    let string_to_print = format!(
498                                        "{:.01}% of {:.02} MB downloading {}",
499                                        bytes_out as f64 / total as f64 * 100.,
500                                        total as f64 / 1000000.,
501                                        crate::shorten_string(path_to_download.as_str(), 80)
502                                    );
503                                    println_to_ui_thread_with_thread_name(&ui_tx, string_to_print, &thread_name);
504                                    just_downloaded = path_to_download.to_string();
505                                } else {
506                                    let string_to_print = format!(
507                                        "{} MB downloaded {}",
508                                        bytes_out as f64 / 1000000.,
509                                        crate::shorten_string(path_to_download.as_str(), 80)
510                                    );
511                                    println_to_ui_thread_with_thread_name(&ui_tx, string_to_print, &thread_name);
512                                }
513                            }
514                            Err(e) => {
515                                let string_to_print = format!("Read error: {}", e);
516                                println_to_ui_thread_with_thread_name(&ui_tx, string_to_print, &thread_name);
517                                continue 'download; // do another request and resume
518                            }
519                        }
520                    }
521                }
522                Ok(Err(download_error)) => {
523                    let string_to_print = format!("Download error: {}", download_error);
524                    println_to_ui_thread_with_thread_name(&ui_tx, string_to_print, &thread_name);
525                }
526                Err(request_error) => {
527                    let string_to_print = format!("Error: {}", request_error);

544    // From this thread I have to send a message to the other thread to avoid multiple threads writing to the same file. That is a no-no.
